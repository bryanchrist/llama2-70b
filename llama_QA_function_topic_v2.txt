total_time = total_distance / speed_of_light

# 2000 miles / 50 miles per minute = 40 minutes
# 40 minutes + 4 * 20 minutes = 100 minutes
# 100 minutes / 60 minutes per hour = 1.67 hours

    # Superman's speed
    speed = 3000
    # Distance to the North Pole
    distance = 2000
    # Time spent in each city
    time_in_city = 20
    # Number of cities
    num_cities = 4

    # Superman's speed in miles per minute
    speed_per_minute = speed / 60

    # Time to fly to the North Pole
    time_to_north_pole = distance / speed_per_minute

    # Time spent in all cities
    time_in_cities = num_cities * time_in_city

    # Total time
    total_time = time_to_north_pole + time_in_cities

    # The answer is
    return total_time
